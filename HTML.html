HyperText Markup Language - Langage de balisage web

Qu'est-ce que le HTML ?

HTML (HyperText Markup Language) est le langage de balisage standard utilisé pour créer et structurer des pages web. Il définit la structure et le contenu d'une page, tandis que CSS s'occupe du style et JavaScript de l'interactivité.

Structure

Définit la structure des pages web

Sémantique

Donne du sens au contenu

Universel

Compris par tous les navigateurs

Structure de base d'un document HTML 

<!DOCTYPE html>
<html lang="fr">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Titre de la page</title>
</head>
<body>
    <h1>Mon premier titre</h1>
    <p>Mon premier paragraphe</p>
</body>
</html>

Balises de texte 

<!-- Titres -->
<h1>Titre principal</h1>
<h2>Sous-titre</h2>
<h3>Titre niveau 3</h3>

<!-- Paragraphes et mise en forme -->
<p>Ceci est un paragraphe normal.</p>
<p>Ce texte est <strong>important</strong></p>
<p>Ce texte est <em>emphasé</em></p>
<p>Ce texte est <u>souligné</u></p>

<!-- Listes -->
<ul>
  <li>Élément 1</li>
  <li>Élément 2</li>
</ul>

<ol>
  <li>Premier</li>
  <li>Deuxième</li>
</ol>

Liens et images

<!-- Liens -->
<a href="https://example.com">Lien externe</a>
<a href="page2.html">Lien interne</a>
<a href="#section1">Lien d'ancrage</a>

<!-- Images -->
<img src="image.jpg" alt="Description de l'image">
<img src="photo.png" alt="Photo" width="300" height="200">

<!-- Figure avec légende -->
<figure>
  <img src="diagramme.svg" alt="Diagramme">
  <figcaption>Diagramme explicatif</figcaption>
</figure>

Formulaires

<form action="/submit" method="POST">
  
  <!-- Champ texte -->
  <label for="nom">Nom :</label>
  <input type="text" id="nom" name="nom" required>
  
  <!-- Email -->
  <label for="email">Email :</label>
  <input type="email" id="email" name="email">
  
  <!-- Mot de passe -->
  <label for="password">Mot de passe :</label>
  <input type="password" id="password" name="password">
  
  <!-- Liste déroulante -->
  <label for="pays">Pays :</label>
  <select id="pays" name="pays">
    <option value="fr">France</option>
    <option value="be">Belgique</option>
  </select>
  
  <!-- Zone de texte -->
  <label for="message">Message :</label>
  <textarea id="message" name="message" rows="4"></textarea>
  
  <!-- Boutons -->
  <button type="submit">Envoyer</button>
  <button type="reset">Effacer</button>
  
</form>

  </nav>

    </section>

    <p>Contenu complémentaire...</p>
  </aside>

Méta-informations importantes 

<head>
  <!-- Encodage -->
  <meta charset="UTF-8">
  
  <!-- Viewport pour le responsive -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  
  <!-- Description -->
  <meta name="description" content="Description de la page">
  
  <!-- Mots-clés -->
  <meta name="keywords" content="HTML, CSS, JavaScript">
  
  <!-- Auteur -->
  <meta name="author" content="Votre Nom">
  
  <!-- Favicon -->
  <link rel="icon" type="image/x-icon" href="/favicon.ico">
  
  <!-- Feuilles de style -->
  <link rel="stylesheet" href="styles.css">
  
  <!-- Scripts -->
  <script src="script.js" defer></script>
</head>

Multimédia

<!-- Audio -->
<audio controls>
  <source src="audio.mp3" type="audio/mpeg">
  <source src="audio.ogg" type="audio/ogg">
  Votre navigateur ne supporte pas l'audio.
</audio>

<!-- Vidéo -->
<video width="320" height="240" controls>
  <source src="video.mp4" type="video/mp4">
  <source src="video.webm" type="video/webm">
  Votre navigateur ne supporte pas la vidéo.
</video>
<!-- Iframe -->
<iframe 
  src="https://example.com" 
  width="600" 
  height="400"
  title="Exemple d'iframe">
</iframe>

<!-- Canvas -->
<canvas id="monCanvas" width="200" height="100">
  Votre navigateur ne supporte pas canvas.
</canvas>

<!-- SVG -->
<svg width="100" height="100">
  <circle cx="50" cy="50" r="40" fill="red" />
</svg>

Attributs globaux importants 

<!-- Classes et ID -->
<div id="header" class="container main">
  <p class="text-primary">Contenu</p>
</div>

<!-- Style inline -->
<p style="color: blue; font-size: 16px;">Texte stylisé</p>

<!-- Données personnalisées -->
<div data-user-id="123" data-role="admin">
  Utilisateur administrateur
</div>

<!-- Accessibilité -->
<button aria-label="Fermer la fenêtre">X</button>
<img src="graph.jpg" alt="Graphique des ventes" 
     aria-describedby="graph-desc">
<p id="graph-desc">Graphique montrant l'évolution des ventes</p>
