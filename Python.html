Fonctions

def greet(name):
    return "Bonjour " + name

print(greet("Alex"))

# Fonction avec paramètre par défaut
def greet_person(name, greeting="Bonjour"):
    return f"{greeting} {name}"

print(greet_person("Marie", "Salut"))

Programmation Orientée Objet 

class User:
    def __init__(self, name, age):
        self.name = name
        self.age = age
    
    def greet(self):
        return f"Bonjour {self.name}"
    
    def is_adult(self):
        return self.age >= 18

user = User("Alex", 20)
print(user.greet())
print(user.is_adult())

Gestion des exceptions 

try:
    number = int(input("Entrez un nombre: "))
    result = 10 / number
    print(f"Résultat: {result}")
except ValueError:
    print("Ce n'est pas un nombre valide!")
except ZeroDivisionError:
    print("Division par zéro impossible!")
except Exception as e:
    print(f"Erreur: {e}")

Exemples avancés 

# List comprehension
squares = [x**2 for x in range(10)]
print(squares)

# Filter avec lambda
numbers = [1, 2, 3, 4, 5, 6]
even_numbers = list(filter(lambda x: x % 2 == 0, numbers))
print(even_numbers)
# Générateur
def fibonacci(limit):
    a, b = 0, 1
    while a < limit:
        yield a
        a, b = b, a + b

for num in fibonacci(20):
    print(num)
