Qu'est-ce que Laravel ?

Laravel est un framework PHP open-source, créé par Taylor Otwell, qui suit le modèle architectural MVC (Model-View-Controller). Il est conçu pour développer des applications web en suivant les bonnes pratiques et en utilisant une syntaxe élégante et expressive.

Architecture Modèle-Vue-Contrôleur

Eloquent ORM

Ecosystème

Forge, Vapor, Nova, Echo

Contrôleurs

<?php
namespace App\Http\Controllers;

use App\Models\User;
use Illuminate\Http\Request;

class UserController extends Controller
{
    // Afficher la liste
    public function index()
    {
        $users = User::all();
        return view('users.index', compact('users'));
    }

    // Afficher un élément
    public function show(User $user)
    {
        return view('users.show', compact('user'));
    }

    // Afficher le formulaire de création
    public function create()
    {
        return view('users.create');
    }

    // Stocker un nouvel élément
    public function store(Request $request)
    {
        $validated = $request->validate([
            'name' => 'required|max:255',
            'email' => 'required|email|unique:users',
        ]);

        User::create($validated);

        return redirect()->route('users.index')
            ->with('success', 'Utilisateur créé avec succès.');
    }

    // Mettre à jour un élément
    public function update(Request $request, User $user)
    {
        $user->update($request->all());
        return redirect()->route('users.index');
    }

    // Supprimer un élément
    public function destroy(User $user)
    {
        $user->delete();
        return redirect()->route('users.index');
    }
}
?>

Eloquent ORM

<?php
// Modèle complet
namespace App\Models;

use Illuminate\Database\Eloquent\Model;
use Illuminate\Database\Eloquent\Factories\HasFactory;

class User extends Model
{
    use HasFactory;

    protected $table = 'users';
    protected $primaryKey = 'id';
    public $timestamps = true;

    protected $fillable = ['name', 'email', 'password'];
    protected $hidden = ['password', 'remember_token'];
    protected $casts = ['email_verified_at' => 'datetime'];

    // Relations
    public function posts()
    {
        return $this->hasMany(Post::class);
    }

    public function profile()
    {
        return $this->hasOne(Profile::class);
    }

    // Scopes
    public function scopeActive($query)
    {
        return $query->where('active', true);
    }

    // Accessors/Mutators
    public function getNameAttribute($value)
    {
        return ucfirst($value);
    }

    public function setPasswordAttribute($value)
    {
        $this->attributes['password'] = bcrypt($value);
    }
}

// Utilisation du modèle
$user = User::find(1);
$activeUsers = User::active()->get();
$user = User::where('email', 'test@example.com')->first();
$newUser = User::create(['name' => 'John', 'email' => 'john@example.com']);
?>

Formulaires et Validation 

<?php
// Validation dans le contrôleur
public function store(Request $request)
{
    $validated = $request->validate([
        'title' => 'required|string|max:255',
        'body' => 'required|string',
        'publish_at' => 'nullable|date',
        'email' => 'required|email|unique:users',
        'password' => 'required|min:8|confirmed',
        'age' => 'required|integer|min:18',
        'website' => 'nullable|url',
        'avatar' => 'image|mimes:jpeg,png,jpg|max:2048',
        'tags' => 'array',
        'tags.*' => 'string|distinct|min:3',
    ], [
        'title.required' => 'Le titre est obligatoire',
        'email.unique' => 'Cet email est déjà utilisé',
    ]);

    // Si la validation échoue, redirection automatique
    Post::create($validated);

    return redirect()->route('posts.index');
}

// Validation manuelle
$validator = Validator::make($request->all(), [
    'name' => 'required',
]);

if ($validator->fails()) {
    return redirect('post/create')
        ->withErrors($validator)
        ->withInput();
}
?>
<?php
// Form Requests
php artisan make:request StorePostRequest

namespace App\Http\Requests;

use Illuminate\Foundation\Http\FormRequest;

class StorePostRequest extends FormRequest
{
    public function authorize()
    {
        return true; // Ou logique métier
    }

    public function rules()
    {
        return [
            'title' => 'required|unique:posts|max:255',
            'body' => 'required',
        ];
    }

    public function messages()
    {
        return [
            'title.required' => 'Un titre est requis',
        ];
    }
}

// Utilisation dans le contrôleur
public function store(StorePostRequest $request)
{
    // La requête est déjà validée
    $validated = $request->validated();
    Post::create($validated);
}
?>
